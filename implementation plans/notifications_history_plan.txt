## 3. Redux State Structure

```typescript
// store/slices/notificationsSlice.ts

import { createSlice, PayloadAction } from '@reduxjs/toolkit';

interface NotificationsState {
  // Active notifications
  active: Notification[];
  
  // Complete history
  history: HistoryEntry[];
  
  // Unread count per player
  unreadCount: Record<string, number>;
  
  // Preferences
  preferences: Record<string, NotificationPreferences>;
  
  // UI state
  ui: {
    showNotificationPanel: boolean;
    showHistoryLog: boolean;
    filter: NotificationFilter;
    expandedNotifications: string[];
  };
  
  // Current turn for history tracking
  currentTurn: number;
}

const initialState: NotificationsState = {
  active: [],
  history: [],
  unreadCount: {},
  preferences: {},
  ui: {
    showNotificationPanel: false,
    showHistoryLog: false,
    filter: {},
    expandedNotifications: [],
  },
  currentTurn: 0,
};

const notificationsSlice = createSlice({
  name: 'notifications',
  initialState,
  reducers: {
    // Add notification
    addNotification: (state, action: PayloadAction<Notification>) => {
      const notification = action.payload;
      
      // Add to active notifications
      state.active.unshift(notification);
      
      // Add to history
      state.history.push({
        id: `history-${notification.id}`,
        turn: state.currentTurn,
        timestamp: notification.timestamp,
        notification,
      });
      
      // Update unread counts
      if (notification.isGlobal) {
        Object.keys(state.unreadCount).forEach(playerId => {
          state.unreadCount[playerId]++;
        });
      } else {
        notification.targetPlayerIds.forEach(playerId => {
          state.unreadCount[playerId] = (state.unreadCount[playerId] || 0) + 1;
        });
      }
      
      // Limit active notifications (keep last 50)
      if (state.active.length > 50) {
        state.active = state.active.slice(0, 50);
      }
    },
    
    // Batch add notifications
    addNotifications: (state, action: PayloadAction<Notification[]>) => {
      action.payload.forEach(notification => {
        state.active.unshift(notification);
        state.history.push({
          id: `history-${notification.id}`,
          turn: state.currentTurn,
          timestamp: notification.timestamp,
          notification,
        });
      });
      
      // Update unread counts
      action.payload.forEach(notification => {
        if (notification.isGlobal) {
          Object.keys(state.unreadCount).forEach(playerId => {
            state.unreadCount[playerId]++;
          });
        } else {
          notification.targetPlayerIds.forEach(playerId => {
            state.unreadCount[playerId] = (state.unreadCount[playerId] || 0) + 1;
          });
        }
      });
      
      if (state.active.length > 50) {
        state.active = state.active.slice(0, 50);
      }
    },
    
    // Mark as read
    markAsRead: (state, action: PayloadAction<{ notificationId: string; playerId: string }>) => {
      const { notificationId, playerId } = action.payload;
      const notification = state.active.find(n => n.id === notificationId);
      
      if (notification && !notification.read) {
        notification.read = true;
        state.unreadCount[playerId] = Math.max(0, (state.unreadCount[playerId] || 0) - 1);
      }
    },
    
    // Mark all as read
    markAllAsRead: (state, action: PayloadAction<string>) => {
      const playerId = action.payload;
      state.active.forEach(notification => {
        if (notification.isGlobal || notification.targetPlayerIds.includes(playerId)) {
          notification.read = true;
        }
      });
      state.unreadCount[playerId] = 0;
    },
    
    // Dismiss notification
    dismissNotification: (state, action: PayloadAction<string>) => {
      const notificationId = action.payload;
      const index = state.active.findIndex(n => n.id === notificationId);
      if (index !== -1) {
        state.active[index].dismissed = true;
      }
    },
    
    // Clear dismissed
    clearDismissed: (state) => {
      state.active = state.active.filter(n => !n.dismissed);
    },
    
    // Clear all notifications
    clearAllNotifications: (state) => {
      state.active = [];
      Object.keys(state.unreadCount).forEach(key => {
        state.unreadCount[key] = 0;
      });
    },
    
    // Update filter
    setFilter: (state, action: PayloadAction<NotificationFilter>) => {
      state.ui.filter = action.payload;
    },
    
    // Toggle UI elements
    toggleNotificationPanel: (state) => {
      state.ui.showNotificationPanel = !state.ui.showNotificationPanel;
    },
    
    toggleHistoryLog: (state) => {
      state.ui.showHistoryLog = !state.ui.showHistoryLog;
    },
    
    // Expand/collapse notification
    toggleNotificationExpand: (state, action: PayloadAction<string>) => {
      const id = action.payload;
      const index = state.ui.expandedNotifications.indexOf(id);
      if (index === -1) {
        state.ui.expandedNotifications.push(id);
      } else {
        state.ui.expandedNotifications.splice(index, 1);
      }
    },
    
    // Update preferences
    updatePreferences: (state, action: PayloadAction<NotificationPreferences>) => {
      const prefs = action.payload;
      state.preferences[prefs.playerId] = prefs;
    },
    
    // Increment turn
    incrementTurn: (state) => {
      state.currentTurn++;
    },
    
    // Initialize player
    initializePlayer: (state, action: PayloadAction<string>) => {
      const playerId = action.payload;
      state.unreadCount[playerId] = 0;
      state.preferences[playerId] = {
        playerId,
        enableSound: true,
        enableVisual: true,
        mutedCategories: [],
        minimumPriority: NotificationPriority.LOW,
      };
    },
    
    // Remove player
    removePlayer: (state, action: PayloadAction<string>) => {
      const playerId = action.payload;
      delete state.unreadCount[playerId];
      delete state.preferences[playerId];
    },
  },
});

export const {
  addNotification,
  addNotifications,
  markAsRead,
  markAllAsRead,
  dismissNotification,
  clearDismissed,
  clearAllNotifications,
  setFilter,
  toggleNotificationPanel,
  toggleHistoryLog,
  toggleNotificationExpand,
  updatePreferences,
  incrementTurn,
  initializePlayer,
  removePlayer,
} = notificationsSlice.actions;

export default notificationsSlice.reducer;
```

## 4. Notification Generator Service

```typescript
// services/NotificationGenerator.ts

import { v4 as uuidv4 } from 'uuid';
import {
  Notification,
  NotificationType,
  NotificationCategory,
  NotificationPriority,
  NotificationData,
} from '@/types/notifications';

class NotificationGenerator {
  private static instance: NotificationGenerator;
  
  private constructor() {}
  
  static getInstance(): NotificationGenerator {
    if (!NotificationGenerator.instance) {
      NotificationGenerator.instance = new NotificationGenerator();
    }
    return NotificationGenerator.instance;
  }
  
  /**
   * Generate notification based on type
   */
  generate(
    type: NotificationType,
    data: NotificationData,
    targetPlayerIds: string[] = []
  ): Notification {
    const template = this.getTemplate(type);
    
    return {
      id: uuidv4(),
      type,
      category: template.category,
      priority: template.priority,
      timestamp: Date.now(),
      title: this.interpolate(template.title, data),
      message: this.interpolate(template.message, data),
      targetPlayerIds,
      isGlobal: targetPlayerIds.length === 0,
      data,
      icon: template.icon,
      color: template.color,
      sound: template.sound,
      read: false,
      dismissed: false,
      expiresAt: template.duration ? Date.now() + template.duration : undefined,
    };
  }
  
  /**
   * Get notification template
   */
  private getTemplate(type: NotificationType): NotificationTemplate {
    return NOTIFICATION_TEMPLATES[type] || DEFAULT_TEMPLATE;
  }
  
  /**
   * Interpolate variables in template strings
   */
  private interpolate(template: string, data: NotificationData): string {
    return template.replace(/\{(\w+)\}/g, (match, key) => {
      return data[key as keyof NotificationData]?.toString() || match;
    });
  }
}
```

## 5. Notification Middleware

```typescript
// store/middleware/notificationMiddleware.ts

import { Middleware } from '@reduxjs/toolkit';
import { addNotification } from '@/store/slices/notificationsSlice';
import NotificationGenerator from '@/services/NotificationGenerator';
import { NotificationType, NotificationData } from '@/types/notifications';
import { playSound } from '@/utils/sound';

/**
 * Middleware to automatically generate notifications from game actions
 */
export const notificationMiddleware: Middleware = (store) => (next) => (action) => {
  const result = next(action);
  
  // Map game actions to notifications
  const notificationMapping = getNotificationMapping(action);
  
  if (notificationMapping) {
    const generator = NotificationGenerator.getInstance();
    const notification = generator.generate(
      notificationMapping.type,
      notificationMapping.data,
      notificationMapping.targetPlayerIds
    );
    
    store.dispatch(addNotification(notification));
    
    // Play sound if enabled
    const state = store.getState();
    const currentPlayer = state.game.currentPlayer;
    const preferences = state.notifications.preferences[currentPlayer?.id];
    
    if (preferences?.enableSound && notification.sound) {
      playSound(notification.sound);
    }
  }
  
  return result;
};

/**
 * Map game actions to notification configurations
 */
function getNotificationMapping(action: any): {
  type: NotificationType;
  data: NotificationData;
  targetPlayerIds: string[];
} | null {
  switch (action.type) {
    // Property actions
    case 'game/buyProperty':
      return {
        type: NotificationType.PROPERTY_PURCHASED,
        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
          propertyId: action.payload.property.id,
          propertyName: action.payload.property.name,
          amount: action.payload.property.price,
        },
        targetPlayerIds: [],
      };
    
    case 'game/startAuction':
      return {
        type: NotificationType.PROPERTY_AUCTIONED,
        data: {
          propertyId: action.payload.property.id,
          propertyName: action.payload.property.name,
        },
        targetPlayerIds: [],
      };
    
    case 'game/auctionWon':
      return {
        type: NotificationType.PROPERTY_AUCTION_WON,
        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
          propertyId: action.payload.property.id,
          propertyName: action.payload.property.name,
          amount: action.payload.amount,
        },
        targetPlayerIds: [],
      };
    
    case 'game/mortgageProperty':
      return {
        type: NotificationType.PROPERTY_MORTGAGED,
        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
          propertyId: action.payload.property.id,
          propertyName: action.payload.property.name,
          amount: action.payload.amount,
        },
        targetPlayerIds: [],
      };
    
    // Building actions
    case 'game/buildHouse':
      return {
        type: NotificationType.HOUSE_BUILT,
        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
          propertyId: action.payload.property.id,
          propertyName: action.payload.property.name,
          buildingCount: action.payload.count,
        },
        targetPlayerIds: [],
      };
    
    case 'game/buildHotel':
      return {
        type: NotificationType.HOTEL_BUILT,
        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
          propertyId: action.payload.property.id,
          propertyName: action.payload.property.name,
        },
        targetPlayerIds: [],
      };
    
    // Rent actions
    case 'game/payRent':
      return {
        type: NotificationType.RENT_PAID,
        data: {
          playerId: action.payload.fromPlayerId,
          playerName: action.payload.toPlayerName,
          propertyId: action.payload.property.id,
          propertyName: action.payload.property.name,
          amount: action.payload.amount,
        },
        targetPlayerIds: [action.payload.fromPlayerId],
      };
    
    // Trade actions
    case 'game/proposeTrade':
      return {
        type: NotificationType.TRADE_PROPOSED,
        data: {
          fromPlayerId: action.payload.fromPlayerId,
          fromPlayerName: action.payload.fromPlayerName,
          toPlayerId: action.payload.toPlayerId,
          toPlayerName: action.payload.toPlayerName,

        },
        targetPlayerIds: [action.payload.toPlayerId],
      };

        type: NotificationType.TRADE_ACCEPTED,
        data: {
          fromPlayerId: action.payload.fromPlayerId,
          toPlayerId: action.payload.toPlayerId,
          toPlayerName: action.payload.toPlayerName,
          tradeId: action.payload.tradeId,
        },
        targetPlayerIds: [action.payload.fromPlayerId],
      };
    
    // Dice roll
    case 'game/rollDice':
      return {
        type: NotificationType.DICE_ROLLED,
        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
          dice1: action.payload.dice1,
          dice2: action.payload.dice2,
          total: action.payload.total,
          isDoubles: action.payload.dice1 === action.payload.dice2,
        },
        targetPlayerIds: [],
      };
    
    // Jail actions
    case 'game/sendToJail':
      return {
        type: NotificationType.SENT_TO_JAIL,
        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
        },
        targetPlayerIds: [],
      };
    
    // Monopoly formed
    case 'game/monopolyFormed':
      return {

        data: {
          playerId: action.payload.playerId,
          playerName: action.payload.playerName,
          colorGroup: action.payload.colorGroup,
```

```typescript
export const selectFilteredNotifications = createSelector(
  [selectActiveNotifications, selectFilter],
  (notifications, filter) => {
    let filtered = notifications;
    
    // Filter by category
    if (filter.categories && filter.categories.length > 0) {
      filtered = filtered.filter((n) => filter.categories!.includes(n.category));
    }
    
    // Filter by priority
    if (filter.priorities && filter.priorities.length > 0) {
      filtered = filtered.filter((n) => filter.priorities!.includes(n.priority));
    }
    
    // Filter by player
    if (filter.playerIds && filter.playerIds.length > 0) {
      filtered = filtered.filter(
        (n) =>
          n.data.playerId && filter.playerIds!.includes(n.data.playerId)
      );
    }
    
    // Filter by time range
    if (filter.startTime) {
      filtered = filtered.filter((n) => n.timestamp >= filter.startTime!);
    }
    if (filter.endTime) {
      filtered = filtered.filter((n) => n.timestamp <= filter.endTime!);
    }
    
    // Filter by search text
    if (filter.searchText) {
      const searchLower = filter.searchText.toLowerCase();
      filtered = filtered.filter(
        (n) =>
          n.title.toLowerCase().includes(searchLower) ||
          n.message.toLowerCase().includes(searchLower)
      );
    }
    
    // Filter by type
    if (filter.types && filter.types.length > 0) {
      filtered = filtered.filter((n) => filter.types!.includes(n.type));
    }
    
    return filtered;
  }
);
```

```typescript
// components/notifications/NotificationToast.tsx

import React, { useEffect, useState } from 'react';
import { useDispatch } from 'react-redux';
import { motion, AnimatePresence } from 'framer-motion';
import { dismissNotification } from '@/store/slices/notificationsSlice';
import { Notification, NotificationPriority } from '@/types/notifications';

interface NotificationToastProps {
  notification: Notification;
}

export const NotificationToast: React.FC<NotificationToastProps> = ({
  notification,
}) => {
  const dispatch = useDispatch();
  const [progress, setProgress] = useState(100);
  
  useEffect(() => {
    if (!notification.expiresAt) return;
    
    const duration = notification.expiresAt - Date.now();
    const interval = 50;
    const decrement = (100 * interval) / duration;
    
    const timer = setInterval(() => {
      setProgress((prev) => {
        if (prev <= 0) {
          clearInterval(timer);
          dispatch(dismissNotification(notification.id));
          return 0;
        }
        return prev - decrement;
      });
    }, interval);
    
    return () => clearInterval(timer);
  }, [notification, dispatch]);
  
  const getPriorityStyles = () => {
    switch (notification.priority) {
      case NotificationPriority.CRITICAL:
        return 'border-red-500 bg-red-50 dark:bg-red-900/20';
      case NotificationPriority.HIGH:
        return 'border-orange-500 bg-orange-50 dark:bg-orange-900/20';
      case NotificationPriority.MEDIUM:
        return 'border-blue-500 bg-blue-50 dark:bg-blue-900/20';
      default:
        return 'border-gray-500 bg-gray-50 dark:bg-gray-900/20';
    }
  };
  
  return (
    <motion.div
      initial={{ opacity: 0, y: -50, scale: 0.9 }}
      animate={{ opacity: 1, y: 0, scale: 1 }}
      exit={{ opacity: 0, x: 300, scale: 0.9 }}
      className={`
        relative overflow-hidden rounded-lg border-l-4 p-4 shadow-lg
        ${getPriorityStyles()}
        min-w-[320px] max-w-[420px]
      `}
    >
      {/* Progress bar */}
      {notification.expiresAt && (
        <div className="absolute bottom-0 left-0 right-0 h-1 bg-gray-200 dark:bg-gray-700">
          <div
            className="h-full bg-blue-500 transition-all duration-50"
            style={{ width: `${progress}%` }}
          />
        </div>
      )}
      
      <div className="flex items-start gap-3">
        {/* Icon */}
        {notification.icon && (
          <div className="text-2xl flex-shrink-0">
            {notification.icon}
          </div>
        )}
        
        {/* Content */}
        <div className="flex-1 min-w-0">
          <h4 className="font-semibold text-gray-900 dark:text-gray-100">
            {notification.title}
          </h4>
          <p className="text-sm text-gray-600 dark:text-gray-300 mt-1">
            {notification.message}
          </p>
        </div>
        
        {/* Close button */}
        <button
          onClick={() => dispatch(dismissNotification(notification.id))}
          className="flex-shrink-0 text-gray-400 hover:text-gray-600 dark:hover:text-gray-200"
        >
          <svg className="w-5 h-5" fill="currentColor" viewBox="0 0 20 20">
            <path
              fillRule="evenodd"
              d="M4.293 4.293a1 1 0 011.414 0L10 8.586l4.293-4.293a1 1 0 111.414 1.414L11.414 10l4.293 4.293a1 1 0 01-1.414 1.414L10 11.414l-4.293 4.293a1 1 0 01-1.414-1.414L8.586 10 4.293 5.707a1 1 0 010-1.414z"
              clipRule="evenodd"
            />
          </svg>
        </button>
      </div>
    </motion.div>
  );
};
```

```typescript
        ) : (

            {notifications.map((notification) => (
              <div
                key={notification.id}
                className={`p-4 hover:bg-gray-50 dark:hover:bg-gray-700/50 cursor-pointer ${
                  !notification.read ? 'bg-blue-50 dark:bg-blue-900/20' : ''
                }`}
                onClick={() => handleMarkAsRead(notification.id)}
              >
                <div className="flex items-start gap-3">
                  {notification.icon && (
                    <div className="text-xl flex-shrink-0">
                      {notification.icon}
                    </div>
                  )}
                  <div className="flex-1 min-w-0">
                    <div className="flex items-start justify-between gap-2">
                      <h4 className="font-medium text-gray-900 dark:text-gray-100 text-sm">
                        {notification.title}
                      </h4>
                      {!notification.read && (
                        <div className="w-2 h-2 bg-blue-500 rounded-full flex-shrink-0 mt-1" />
                      )}
                    </div>
                    <p className="text-sm text-gray-600 dark:text-gray-300 mt-1">
                      {notification.message}
                    </p>
                    <p className="text-xs text-gray-400 dark:text-gray-500 mt-2">
                      {formatTimestamp(notification.timestamp)}
                    </p>
                  </div>
                </div>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
};

function formatTimestamp(timestamp: number): string {
  const now = Date.now();
  const diff = now - timestamp;
  
  if (diff < 60000) return 'Just now';
  if (diff < 3600000) return `${Math.floor(diff / 60000)}m ago`;
  if (diff < 86400000) return `${Math.floor(diff / 3600000)}h ago`;
  return new Date(timestamp).toLocaleDateString();
}
```

```typescript
// utils/sound.ts

class SoundManager {
  private static instance: SoundManager;
  private sounds: Map<string, HTMLAudioElement> = new Map();
  private enabled: boolean = true;
  
  private constructor() {
    this.loadSounds();
  }
  
  static getInstance(): SoundManager {
    if (!SoundManager.instance) {
      SoundManager.instance = new SoundManager();
    }
    return SoundManager.instance;
  }
  
  private loadSounds() {
    const soundFiles: Record<string, string> = {
      purchase: '/sounds/purchase.mp3',
      auction: '/sounds/auction.mp3',
      win: '/sounds/win.mp3',
      mortgage: '/sounds/mortgage.mp3',
      build: '/sounds/build.mp3',
      hotel: '/sounds/hotel.mp3',
      sell: '/sounds/sell.mp3',
      payment: '/sounds/payment.mp3',
      receive: '/sounds/receive.mp3',
      trade_offer: '/sounds/trade_offer.mp3',
      trade_accept: '/sounds/trade_accept.mp3',
      trade_reject: '/sounds/trade_reject.mp3',
      salary: '/sounds/salary.mp3',
      tax: '/sounds/tax.mp3',
      jail: '/sounds/jail.mp3',
      card: '/sounds/card.mp3',
      dice: '/sounds/dice.mp3',
      doubles: '/sounds/doubles.mp3',
      bankruptcy: '/sounds/bankruptcy.mp3',
      monopoly: '/sounds/monopoly.mp3',
      start: '/sounds/start.mp3',
      victory: '/sounds/victory.mp3',
    };
    
    Object.entries(soundFiles).forEach(([key, path]) => {
      const audio = new Audio(path);
      audio.preload = 'auto';
      this.sounds.set(key, audio);
    });
  }
  
  play(soundName: string) {
    if (!this.enabled) return;
    
    const sound = this.sounds.get(soundName);
    if (sound) {
      sound.currentTime = 0;
      sound.play().catch((err) => console.error('Error playing sound:', err));
    }
  }
  
  setEnabled(enabled: boolean) {
    this.enabled = enabled;
  }
  
  isEnabled(): boolean {
    return this.enabled;
  }
}

export const playSound = (soundName: string) => {
  SoundManager.getInstance().play(soundName);
};

export const setSoundEnabled = (enabled: boolean) => {
  SoundManager.getInstance().setEnabled(enabled);
};
```

```typescript
// hooks/useNotifications.ts

import { useDispatch } from 'react-redux';
import { useCallback } from 'react';
import NotificationGenerator from '@/services/NotificationGenerator';
import { addNotification } from '@/store/slices/notificationsSlice';
import { NotificationType, NotificationData } from '@/types/notifications';

export const useNotifications = () => {
  const dispatch = useDispatch();
  
  const notify = useCallback(
    (type: NotificationType, data: NotificationData, targetPlayerIds: string[] = []) => {
      const generator = NotificationGenerator.getInstance();
      const notification = generator.generate(type, data, targetPlayerIds);
      dispatch(addNotification(notification));
    },
    [dispatch]
  );
  
  return { notify };
};
```

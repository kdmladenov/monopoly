### 2.3 Game Engine Implementation
```typescript
// services/GameEngine.ts

class GameEngine {
  private store: Store;
  private eventEmitter: EventEmitter;
  
  constructor(store: Store) {
    this.store = store;
    this.eventEmitter = new EventEmitter();
  }
  
  /**
   * Initialize a new game
   */
  initializeGame(config: GameConfig, players: Player[]): void {
    // Validate configuration
    this.validateGameConfig(config);
    
    // Initialize board
    const board = this.initializeBoard();
    
    // Shuffle card decks
    const { chanceDeck, communityChestDeck } = this.shuffleDecks();
    
    // Set starting positions
    players.forEach(player => {
      player.position = 0;
      player.cash = config.startingCash;
    });
    
    // Determine turn order
    const turnOrder = this.determineTurnOrder(players);
    
    // Dispatch initialization
    this.store.dispatch(
      gameActions.initialize({
        config,
        board,
        players,
        turnOrder,
        chanceDeck,
        communityChestDeck,
      })
    );
    
    this.emit('game:initialized');
  }
  
  /**
   * Start the game
   */
  startGame(): void {
    const state = this.store.getState();
    
    if (state.game.status !== GameStatus.LOBBY) {
      throw new Error('Cannot start game from current status');
    }
    
    // Set first player
    const firstPlayer = state.players.turnOrder[0];
    
    this.store.dispatch(
      gameActions.start({
        currentPlayerId: firstPlayer,
        startTime: Date.now(),
      })
    );
    
    this.emit('game:started');
    this.startTurn(firstPlayer);
  }
  
  /**
   * Process a turn
   */
  startTurn(playerId: string): void {
    const state = this.store.getState();
    const player = state.players.entities[playerId];
    
    // Check if player is bankrupt
    if (player.isBankrupt) {
      this.nextTurn();
      return;
    }
    
    // Initialize turn state
    this.store.dispatch(
      gameActions.startTurn({
        playerId,
        turnStartTime: Date.now(),
      })
    );
    
    // Check if player is in jail
    if (player.isInJail) {
      this.handleJailTurn(playerId);
    } else {
      this.setPhase(GamePhase.ROLL_DICE);
    }
    
    this.emit('turn:started', { playerId });
  }
  
  /**
   * Roll dice
   */
  rollDice(playerId: string): DiceRoll {
    const state = this.store.getState();
    
    // Validate can roll
    if (!this.canRollDice(playerId)) {
      throw new Error('Cannot roll dice at this time');
    }
    
    const dice1 = this.randomDice();
    const dice2 = this.randomDice();
    const total = dice1 + dice2;
    const isDoubles = dice1 === dice2;
    
    const roll: DiceRoll = {
      dice1,
      dice2,
      total,
      isDoubles,
      playerId,
      timestamp: Date.now(),
    };
    
    this.store.dispatch(gameActions.rollDice(roll));
    
    // Handle doubles
    if (isDoubles) {
      this.handleDoubles(playerId);
    }
    
    // Move player
    this.movePlayer(playerId, total);
    
    this.emit('dice:rolled', roll);
    
    return roll;
  }
  
  /**
   * Move player
   */
  movePlayer(playerId: string, spaces: number): void {
    const state = this.store.getState();
    const player = state.players.entities[playerId];
    const oldPosition = player.position;
    let newPosition = (oldPosition + spaces) % 40;
    
    // Check if passed GO
    const passedGo = newPosition < oldPosition || (oldPosition + spaces >= 40);
    
    if (passedGo) {
      this.collectSalary(playerId);
    }
    
    // Update position
    this.store.dispatch(
      playersActions.updatePosition({
        playerId,
        oldPosition,
        newPosition,
        passedGo,
      })
    );
    
    this.emit('player:moved', { playerId, oldPosition, newPosition, passedGo });
    
    // Handle landing
    this.handleLanding(playerId, newPosition);
  }
  
  /**
   * Handle landing on a space
   */
  handleLanding(playerId: string, position: number): void {
    const state = this.store.getState();
    const space = state.game.board.spaces[position];
    
    this.setPhase(GamePhase.LAND_ON_SPACE);
    
    switch (space.type) {
      case SpaceType.PROPERTY:
        this.handlePropertyLanding(playerId, space as PropertySpace);
        break;
        
      case SpaceType.RAILROAD:
        this.handleRailroadLanding(playerId, space as RailroadSpace);
        break;
        
      case SpaceType.UTILITY:
        this.handleUtilityLanding(playerId, space as UtilitySpace);
        break;
        
      case SpaceType.CHANCE:
        this.drawChanceCard(playerId);
        break;
        
      case SpaceType.COMMUNITY_CHEST:
        this.drawCommunityChestCard(playerId);
        break;
        
      case SpaceType.TAX:
        this.payTax(playerId, space as TaxSpace);
        break;
        
      case SpaceType.GO_TO_JAIL:
        this.sendToJail(playerId);
        break;
        
      case SpaceType.FREE_PARKING:
        this.handleFreeParking(playerId);
        break;
        
      case SpaceType.GO:
      case SpaceType.JAIL:
        // Just visiting, no action
        this.endTurn();
        break;
    }
    
    this.emit('space:landed', { playerId, space });
  }
  
  /**
   * Execute a game action
   */
  executeAction(action: GameAction): GameResult {
    // Validate action
    const validation = this.validateAction(action);
    
    if (!validation.valid) {
      return {
        success: false,
        error: validation.error,
      };
    }
    
    // Execute based on action type
    switch (action.type) {
      case GameActionType.BUY_PROPERTY:
        return this.buyProperty(action.playerId, action.propertyId);
        
      case GameActionType.AUCTION_PROPERTY:
        return this.startAuction(action.propertyId);
        
      case GameActionType.BUILD_HOUSE:
        return this.buildHouse(action.playerId, action.propertyId);
        
      case GameActionType.BUILD_HOTEL:
        return this.buildHotel(action.playerId, action.propertyId);
        
      case GameActionType.SELL_HOUSE:
        return this.sellHouse(action.playerId, action.propertyId);
        
      case GameActionType.MORTGAGE_PROPERTY:
        return this.mortgageProperty(action.playerId, action.propertyId);
        
      case GameActionType.UNMORTGAGE_PROPERTY:
        return this.unmortgageProperty(action.playerId, action.propertyId);
        
      case GameActionType.PROPOSE_TRADE:
        return this.proposeTrade(action.trade);
        
      case GameActionType.ACCEPT_TRADE:
        return this.acceptTrade(action.tradeId);
        
      case GameActionType.REJECT_TRADE:
        return this.rejectTrade(action.tradeId);
        
      case GameActionType.PAY_JAIL_FINE:
        return this.payJailFine(action.playerId);
        
      case GameActionType.USE_JAIL_CARD:
        return this.useGetOutOfJailCard(action.playerId);
        
      case GameActionType.DECLARE_BANKRUPTCY:
        return this.declareBankruptcy(action.playerId);
        
      case GameActionType.END_TURN:
        return this.endTurn();
        
      default:
        return {
          success: false,
          error: 'Unknown action type',
        };
    }
  }
  
  /**
   * Validate an action
   */
  validateAction(action: GameAction): ValidationResult {
    const state = this.store.getState();
    
    // Check if it's the player's turn
    if (action.playerId !== state.game.currentPlayerId) {
      return {
        valid: false,
        error: 'Not your turn',
      };
    }
    
    // Action-specific validation
    switch (action.type) {
      case GameActionType.BUY_PROPERTY:
        return this.validateBuyProperty(action);
        
      case GameActionType.BUILD_HOUSE:
        return this.validateBuildHouse(action);
        
      // ... other validations
        
      default:
        return { valid: true };
    }
  }
  
  /**
   * End current turn
   */
  endTurn(): GameResult {
    const state = this.store.getState();
    const currentPlayerId = state.game.currentPlayerId;
    const turnState = state.game.turnState;
    
    // Check if player can roll again (doubles)
    if (turnState.canRollAgain && !turnState.hasMovedThisTurn) {
      return {
        success: false,
        error: 'Must roll again (doubles)',
      };
    }
    
    // Clear turn state
    this.store.dispatch(gameActions.endTurn());
    
    this.emit('turn:ended', { playerId: currentPlayerId });
    
    // Move to next player
    this.nextTurn();
    
    return { success: true };
  }
  
  /**
   * Move to next player
   */
  nextTurn(): void {
    const state = this.store.getState();
    const turnOrder = state.players.turnOrder;
    const currentIndex = turnOrder.indexOf(state.game.currentPlayerId);
    const nextIndex = (currentIndex + 1) % turnOrder.length;
    const nextPlayerId = turnOrder[nextIndex];
    
    // Increment round if completed full rotation
    if (nextIndex === 0) {
      this.store.dispatch(gameActions.incrementRound());
    }
    
    // Check for game end condition
    if (this.checkGameEnd()) {
      this.endGame();
      return;
    }
    
    // Start next turn
    this.startTurn(nextPlayerId);
  }
  
  /**
   * Check if game should end
   */
  checkGameEnd(): boolean {
    const state = this.store.getState();
    const activePlayers = state.players.ids.filter(
      id => !state.players.entities[id].isBankrupt
    );
    
    return activePlayers.length === 1;
  }
  
  /**
   * End the game
   */
  endGame(): void {
    const state = this.store.getState();
    const activePlayers = state.players.ids.filter(
      id => !state.players.entities[id].isBankrupt
    );
    
    const winnerId = activePlayers[0];
    
    this.store.dispatch(
      gameActions.endGame({
        winnerId,
        endTime: Date.now(),
        finalStandings: this.calculateFinalStandings(),
      })
    );
    
    this.emit('game:ended', { winnerId });
  }
  
  // Utility methods
  private randomDice(): number {
    return Math.floor(Math.random() * 6) + 1;
  }
  
  private emit(event: string, data?: any): void {
    this.eventEmitter.emit(event, data);
  }
  
  on(event: string, handler: Function): void {
    this.eventEmitter.on(event, handler);
  }
}

export default GameEngine;
```

### 4.2 Players Redux Slice
```typescript
// store/slices/playersSlice.ts

import { createSlice, createEntityAdapter, PayloadAction } from '@reduxjs/toolkit';

const playersAdapter = createEntityAdapter<Player>();

interface PlayersState extends ReturnType<typeof playersAdapter.getInitialState> {
  turnOrder: string[];
  eliminatedOrder: string[];
}

const initialState: PlayersState = playersAdapter.getInitialState({
  turnOrder: [],
  eliminatedOrder: [],
});

const playersSlice = createSlice({
  name: 'players',
  initialState,
  reducers: {
    addPlayer: (state, action: PayloadAction<Player>) => {
      playersAdapter.addOne(state, action.payload);
      state.turnOrder.push(action.payload.id);
    },
    
    removePlayer: (state, action: PayloadAction<string>) => {
      const playerId = action.payload;
      playersAdapter.removeOne(state, playerId);
      state.turnOrder = state.turnOrder.filter(id => id !== playerId);
    },
    
    updatePlayer: (state, action: PayloadAction<{ id: string; changes: Partial<Player> }>) => {
      playersAdapter.updateOne(state, {
        id: action.payload.id,
        changes: action.payload.changes,
      });
    },
    
    updateCash: (state, action: PayloadAction<{ playerId: string; amount: number }>) => {
      const { playerId, amount } = action.payload;
      const player = state.entities[playerId];
      if (player) {
        player.cash += amount;
        player.netWorth += amount;
      }
    },
    
    setCash: (state, action: PayloadAction<{ playerId: string; cash: number }>) => {
      const { playerId, cash } = action.payload;
      const player = state.entities[playerId];
      if (player) {
        const diff = cash - player.cash;
        player.cash = cash;
        player.netWorth += diff;
      }
    },
    
    updatePosition: (state, action: PayloadAction<{
      playerId: string;
      position: number;
      passedGo?: boolean;
    }>) => {
      const { playerId, position } = action.payload;
      const player = state.entities[playerId];
      if (player) {
        player.position = position;
      }
    },
    
    sendToJail: (state, action: PayloadAction<string>) => {
      const player = state.entities[action.payload];
      if (player) {
        player.isInJail = true;
        player.jailTurns = 0;
        player.position = 10; // Jail position
        player.stats.timesInJail++;
      }
    },
    
    releaseFromJail: (state, action: PayloadAction<string>) => {
      const player = state.entities[action.payload];
      if (player) {
        player.isInJail = false;
        player.jailTurns = 0;
        player.doublesCount = 0;
      }
    },
    
    incrementJailTurns: (state, action: PayloadAction<string>) => {
      const player = state.entities[action.payload];
      if (player) {
        player.jailTurns++;
      }
    },
    
    addProperty: (state, action: PayloadAction<{ playerId: string; propertyId: string }>) => {
      const { playerId, propertyId } = action.payload;
      const player = state.entities[playerId];
      if (player && !player.propertyIds.includes(propertyId)) {
        player.propertyIds.push(propertyId);
      }
    },
    
    removeProperty: (state, action: PayloadAction<{ playerId: string; propertyId: string }>) => {
      const { playerId, propertyId } = action.payload;
      const player = state.entities[playerId];
      if (player) {
        player.propertyIds = player.propertyIds.filter(id => id !== propertyId);
      }
    },
    
    declareBankruptcy: (state, action: PayloadAction<{ playerId: string; turn: number }>) => {
      const { playerId, turn } = action.payload;
      const player = state.entities[playerId];
      if (player) {
        player.isBankrupt = true;
        player.isActive = false;
        player.bankruptTurn = turn;
        state.eliminatedOrder.push(playerId);
      }
    },
    
    updateStats: (state, action: PayloadAction<{
      playerId: string;
      stats: Partial<PlayerStats>;
    }>) => {
      const { playerId, stats } = action.payload;
      const player = state.entities[playerId];
      if (player) {
        Object.assign(player.stats, stats);
      }
    },
    
    setTurnOrder: (state, action: PayloadAction<string[]>) => {
      state.turnOrder = action.payload;
    },
  },
});

export const {
  addPlayer,
  removePlayer,
  updatePlayer,
  updateCash,
  setCash,
  updatePosition,
  sendToJail,
  releaseFromJail,
  incrementJailTurns,
  addProperty,
  removeProperty,
  declareBankruptcy,
  updateStats,
  setTurnOrder,
} = playersSlice.actions;

export default playersSlice.reducer;

// Selectors
export const playersSelectors = playersAdapter.getSelectors<RootState>(
  (state) => state.players
);

export const selectActivePlayers = (state: RootState) =>
  playersSelectors.selectAll(state).filter(p => !p.isBankrupt);

export const selectCurrentPlayer = (state: RootState) => {
  const currentId = state.game.currentPlayerId;
  return currentId ? playersSelectors.selectById(state, currentId) : null;
};

export const selectPlayersByNetWorth = (state: RootState) =>
  playersSelectors
    .selectAll(state)
    .sort((a, b) => b.netWorth - a.netWorth);
```

```typescript
import { createSlice, createEntityAdapter } from '@reduxjs/toolkit';

const propertiesAdapter = createEntityAdapter<Property>();

const propertiesSlice = createSlice({
  name: 'properties',
  initialState: propertiesAdapter.getInitialState(),
  reducers: {
    initializeProperties: (state, action: PayloadAction<Property[]>) => {
      propertiesAdapter.setAll(state, action.payload);
    },
    
    setOwner: (state, action: PayloadAction<{ propertyId: string; ownerId: string }>) => {
      const { propertyId, ownerId } = action.payload;
      const property = state.entities[propertyId];
      if (property) {
        property.ownerId = ownerId;
      }
    },
    
    clearOwner: (state, action: PayloadAction<string>) => {
      const property = state.entities[action.payload];
      if (property) {
        property.ownerId = undefined;
      }
    },
    
    mortgageProperty: (state, action: PayloadAction<string>) => {
      const property = state.entities[action.payload];
      if (property) {
        property.isMortgaged = true;
      }
    },
    
    unmortgageProperty: (state, action: PayloadAction<string>) => {
      const property = state.entities[action.payload];
      if (property) {
        property.isMortgaged = false;
      }
    },
    
    addBuilding: (state, action: PayloadAction<{ propertyId: string; count: number }>) => {
      const { propertyId, count } = action.payload;
      const property = state.entities[propertyId] as StreetProperty;
      if (property && property.type === PropertyType.STREET) {
        property.buildings = Math.min(5, property.buildings + count);
      }
    },
    
    removeBuilding: (state, action: PayloadAction<{ propertyId: string; count: number }>) => {
      const { propertyId, count } = action.payload;
      const property = state.entities[propertyId] as StreetProperty;
      if (property && property.type === PropertyType.STREET) {
        property.buildings = Math.max(0, property.buildings - count);
      }
    },
    
    transferProperty: (state, action: PayloadAction<{
      propertyId: string;
      fromPlayerId: string;
      toPlayerId: string;
    }>) => {
      const { propertyId, toPlayerId } = action.payload;
      const property = state.entities[propertyId];
      if (property) {
        property.ownerId = toPlayerId;
      }
    },
  },
});

export const selectCanBuildOn = (propertyId: string) => (state: RootState) => {
  const property = propertiesSelectors.selectById(state, propertyId) as StreetProperty;
  
  if (!property || property.type !== PropertyType.STREET || !property.ownerId) {
    return false;
  }
  
  // Must own monopoly
  const monopolies = selectPlayerMonopolies(property.ownerId)(state);
  if (!monopolies.includes(property.colorGroup)) {
    return false;
  }
  
  // Cannot be mortgaged
  if (property.isMortgaged) {
    return false;
  }
  
  // Even building rule
  const groupProperties = selectPropertiesByColorGroup(property.colorGroup)(state) as StreetProperty[];
  const minBuildings = Math.min(...groupProperties.map(p => p.buildings));
  const maxBuildings = Math.max(...groupProperties.map(p => p.buildings));
  
  // Can't build if this property already has the most buildings
  if (property.buildings === maxBuildings && maxBuildings > minBuildings) {
    return false;
  }
  
  return property.buildings < 5;
};
```

```typescript
// services/BuildingService.ts

class BuildingService {
  /**
   * Validate building action
   */
  canBuild(
    playerId: string,
    propertyId: string,
    count: number,
    state: RootState
  ): ValidationResult {
    const property = state.properties.entities[propertyId] as StreetProperty;
    const player = state.players.entities[playerId];
    
    if (!property || !player) {
      return { valid: false, error: 'Invalid property or player' };
    }
    
    // Must be a street property
    if (property.type !== PropertyType.STREET) {
      return { valid: false, error: 'Can only build on street properties' };
    }
    
    // Must own the property
    if (property.ownerId !== playerId) {
      return { valid: false, error: 'You do not own this property' };
    }
    
    // Must own all properties in color group
    const monopolies = selectPlayerMonopolies(playerId)(state);
    if (!monopolies.includes(property.colorGroup)) {
      return { valid: false, error: 'Must own complete color group' };
    }
    
    // Cannot be mortgaged
    if (property.isMortgaged) {
      return { valid: false, error: 'Cannot build on mortgaged property' };
    }
    
    // Check even building rule
    const groupProperties = selectPropertiesByColorGroup(property.colorGroup)(state) as StreetProperty[];
    const buildingCounts = groupProperties.map(p => p.buildings);
    const minBuildings = Math.min(...buildingCounts);
    const maxBuildings = Math.max(...buildingCounts);
    
    if (property.buildings === maxBuildings && maxBuildings > minBuildings) {
      return { valid: false, error: 'Must build evenly across color group' };
    }
    
    // Check building limits
    const newBuildingCount = property.buildings + count;
    if (newBuildingCount > 5) {
      return { valid: false, error: 'Cannot exceed 5 buildings (1 hotel)' };
    }
    
    // Check available houses/hotels
    if (newBuildingCount === 5 && count === 1) {
      // Building hotel
      if (state.game.availableHotels < 1) {
        return { valid: false, error: 'No hotels available' };
      }
    } else {
      // Building houses
      if (state.game.availableHouses < count) {
        return { valid: false, error: `Only ${state.game.availableHouses} houses available` };
      }
    }
    
    // Check if player can afford it
    const cost = property.houseCost * count;
    if (player.cash < cost) {
      return { valid: false, error: 'Insufficient funds' };
    }
    
    return { valid: true };
  }
  
  /**
   * Build houses/hotel
   */
  build(
    playerId: string,
    propertyId: string,
    count: number,
    dispatch: AppDispatch
  ): GameResult {
    // Validation is already done
    const property = store.getState().properties.entities[propertyId] as StreetProperty;
    const cost = property.houseCost * count;
    
    // Deduct cash
    dispatch(updateCash({ playerId, amount: -cost }));
    
    // Add buildings
    dispatch(addBuilding({ propertyId, count }));
    
    // Update available houses/hotels
    if (property.buildings + count === 5) {
      dispatch(gameActions.useHotel());
      dispatch(gameActions.returnHouses(4)); // Return 4 houses when building hotel
    } else {
      dispatch(gameActions.useHouses(count));
    }
    
    // Update stats
    dispatch(updateStats({
      playerId,
      stats: {
        housesBuilt: property.buildings + count < 5 ? count : 0,
        hotelsBuilt: property.buildings + count === 5 ? 1 : 0,
      },
    }));
    
    return { success: true };
  }
  
  /**
   * Sell buildings
   */
  sell(
    playerId: string,
    propertyId: string,
    count: number,
    dispatch: AppDispatch
  ): GameResult {
    const property = store.getState().properties.entities[propertyId] as StreetProperty;
    
    // Selling gives back half the building cost
    const returnValue = (property.houseCost * count) / 2;
    
    // Add cash
    dispatch(updateCash({ playerId, amount: returnValue }));
    
    // Remove buildings
    dispatch(removeBuilding({ propertyId, count }));
    
    // Return houses/hotels to bank
    if (property.buildings === 5 && count >= 1) {
      // Selling hotel
      dispatch(gameActions.returnHotel());
      if (count > 1) {
        dispatch(gameActions.returnHouses(count - 1));
      }
    } else {
      dispatch(gameActions.returnHouses(count));
    }
    
    return { success: true, data: { returnValue } };
  }
}
```

```typescript
export const DiceRoller: React.FC = () => {

  const [dice1, setDice1] = useState(1);

    if (!canRoll || !currentPlayer) return;
    
    setIsRolling(true);
    
    // Animate dice rolling
    const rollDuration = 1000;
    const rollInterval = 100;
    const rollAnimation = setInterval(() => {
      setDice1(Math.floor(Math.random() * 6) + 1);
      setDice2(Math.floor(Math.random() * 6) + 1);
    }, rollInterval);
    
    // Dispatch roll action
    setTimeout(() => {
      clearInterval(rollAnimation);
      dispatch(gameActions.rollDice({ playerId: currentPlayer.id }));
      setIsRolling(false);
    }, rollDuration);
  };
  
  useEffect(() => {
    if (lastRoll) {
      setDice1(lastRoll.dice1);
      setDice2(lastRoll.dice2);
    }
  }, [lastRoll]);
  
  return (
    <div className="dice-roller flex flex-col items-center gap-4">
      {/* Dice display */}
      <div className="flex gap-4">
        <Die value={dice1} isRolling={isRolling} />
```

```typescript
// services/JailService.ts

class JailService {
  /**
   * Send player to jail
   */
  sendToJail(playerId: string, dispatch: AppDispatch): void {
    dispatch(sendToJail(playerId));
    dispatch(updatePosition({ playerId, position: 10 }));
    
    // Reset doubles count
    dispatch(gameActions.resetDoubles(playerId));
  }
  
  /**
   * Handle jail turn
   */
  handleJailTurn(playerId: string, state: RootState, dispatch: AppDispatch): void {
    const player = state.players.entities[playerId];
    
    if (!player || !player.isInJail) return;
    
    // Check if max turns reached (auto-pay fine)
    if (player.jailTurns >= 2) {
      this.payJailFine(playerId, dispatch);
      return;
    }
    
    // Present options to player
    dispatch(uiActions.showJailOptions(playerId));
  }
  
  /**
   * Pay jail fine
   */
  payJailFine(playerId: string, dispatch: AppDispatch): GameResult {
    const state = store.getState();
    const player = state.players.entities[playerId];
    const config = state.game.config;
    
    if (!player) {
      return { success: false, error: 'Player not found' };
    }
    
    if (player.cash < config.jailFine) {
      return { success: false, error: 'Insufficient funds' };
    }
    
    // Pay fine
    dispatch(updateCash({ playerId, amount: -config.jailFine }));
    
    // Release from jail
    dispatch(releaseFromJail(playerId));
    
    // Update stats
    dispatch(updateStats({
      playerId,
      stats: { jailFinesPaid: 1 },
    }));
    
    return { success: true };
  }
  
  /**
   * Use Get Out of Jail Free card
   */
  useJailCard(playerId: string, dispatch: AppDispatch): GameResult {
    const state = store.getState();
    const player = state.players.entities[playerId];
    
    if (!player || player.getOutOfJailCards === 0) {
      return { success: false, error: 'No card available' };
    }
    
    // Use card
    dispatch(playersActions.useGetOutOfJailCard(playerId));
    
    // Release from jail
    dispatch(releaseFromJail(playerId));
    
    return { success: true };
  }
  
  /**
   * Attempt to roll doubles in jail
   */
  rollForDoubles(playerId: string, dispatch: AppDispatch): GameResult {
    const dice1 = Math.floor(Math.random() * 6) + 1;
    const dice2 = Math.floor(Math.random() * 6) + 1;
    const isDoubles = dice1 === dice2;
    
    dispatch(gameActions.rollDice({
      playerId,
      dice1,
      dice2,
      total: dice1 + dice2,
      isDoubles,
    }));
    
    if (isDoubles) {
      // Release from jail
      dispatch(releaseFromJail(playerId));
      
      // Move player
      dispatch(gameActions.movePlayer({
        playerId,
        spaces: dice1 + dice2,
      }));
      
      return { success: true, data: { released: true } };
    } else {
      // Increment jail turns
      dispatch(incrementJailTurns(playerId));
      
      return { success: true, data: { released: false } };
    }
  }
}
```

```typescript
// services/CardService.ts

class CardService {
  /**
   * Draw a card
   */
  drawCard(cardType: CardType, playerId: string, dispatch: AppDispatch): Card {
    const state = store.getState();
    const deck = cardType === CardType.CHANCE 
      ? state.game.board.chanceCardDeck
      : state.game.board.communityChestDeck;
    
    // Get top card
    const cardId = deck[0];
    const card = cardType === CardType.CHANCE
      ? state.game.board.chanceCards.find(c => c.id === cardId)
      : state.game.board.communityChestCards.find(c => c.id === cardId);
    
    if (!card) {
      throw new Error('No cards in deck');
    }
    
    // Move card to discard pile (unless it's Get Out of Jail Free)
    if (card.action.type !== 'GET_OUT_OF_JAIL') {
      dispatch(gameActions.moveCardToDiscard({ cardType, cardId }));
    } else {
      // Give card to player
      dispatch(playersActions.giveGetOutOfJailCard(playerId));
      dispatch(gameActions.removeCardFromDeck({ cardType, cardId }));
    }
    
    // Execute card action
    this.executeCardAction(card, playerId, dispatch);
    
    return card;
  }
  
  /**
   * Execute card action
   */
  private executeCardAction(card: Card, playerId: string, dispatch: AppDispatch): void {
    const action = card.action;
    
    switch (action.type) {
      case 'MONEY':
        dispatch(updateCash({ playerId, amount: action.amount }));
        break;
        
      case 'MOVE':
        this.handleMoveAction(action, playerId, dispatch);
        break;
        
      case 'GET_OUT_OF_JAIL':
        // Already handled in drawCard
        break;
        
      case 'REPAIRS':
        this.handleRepairsAction(action, playerId, dispatch);
        break;
        
      case 'PAY_PLAYERS':
        this.handlePayPlayersAction(action, playerId, dispatch);
        break;
    }
  }
  
  private handleMoveAction(action: MoveAction, playerId: string, dispatch: AppDispatch): void {
    const state = store.getState();
    const player = state.players.entities[playerId];
    
    if (!player) return;
    
    let destination: number;
    
    if (typeof action.destination === 'number') {
      destination = action.destination;
    } else if (action.destination === 'BACK') {
      destination = (player.position + (action.spaces || 0) + 40) % 40;
    } else if (action.destination === 'JAIL') {
      dispatch(sendToJail(playerId));
      return;
    } else if (action.destination === 'NEAREST_RAILROAD') {
      const railroads = [5, 15, 25, 35];
      destination = this.findNearest(player.position, railroads);
    } else if (action.destination === 'NEAREST_UTILITY') {
      const utilities = [12, 28];
      destination = this.findNearest(player.position, utilities);
    } else {
      destination = 0; // GO
    }
    
    const passedGo = destination < player.position || (action.collectGo && destination !== 0);
    
    if (passedGo && action.collectGo) {
      dispatch(updateCash({ playerId, amount: state.game.config.salaryAmount }));
    }
    
    dispatch(updatePosition({ playerId, position: destination, passedGo }));
    
    // Handle landing
    dispatch(gameActions.handleLanding({ playerId, position: destination }));
  }
  
  private handleRepairsAction(action: RepairsAction, playerId: string, dispatch: AppDispatch): void {
    const state = store.getState();
    const player = state.players.entities[playerId];
    
    if (!player) return;
    
    let totalCost = 0;
    
    player.propertyIds.forEach(propId => {
      const property = state.properties.entities[propId] as StreetProperty;
      if (property && property.type === PropertyType.STREET) {
        if (property.buildings === 5) {
          totalCost += action.perHotel;
        } else {
          totalCost += property.buildings * action.perHouse;
        }
      }
    });
    
    dispatch(updateCash({ playerId, amount: -totalCost }));
  }
  
  private handlePayPlayersAction(action: PayPlayersAction, playerId: string, dispatch: AppDispatch): void {
    const state = store.getState();
    const activePlayers = state.players.ids.filter(id => 
      id !== playerId && !state.players.entities[id].isBankrupt
    );
    
    const totalAmount = action.amountPerPlayer * activePlayers.length;
    
    // Deduct from current player
    dispatch(updateCash({ playerId, amount: -totalAmount }));
    
    // Give to each player
    activePlayers.forEach(id => {
      dispatch(updateCash({ playerId: id, amount: action.amountPerPlayer }));
    });
  }
  
  private findNearest(currentPosition: number, positions: number[]): number {
    for (const pos of positions) {
      if (pos > currentPosition) {
        return pos;
      }
    }
    return positions[0]; // Wrap around
  }
  
  /**
   * Shuffle deck
   */
  shuffleDeck(cardType: CardType, dispatch: AppDispatch): void {
    const state = store.getState();
    const deck = cardType === CardType.CHANCE
      ? [...state.game.board.chanceCardDeck, ...state.game.board.chanceCardDiscardPile]
      : [...state.game.board.communityChestDeck, ...state.game.board.communityChestDiscardPile];
    
    // Fisher-Yates shuffle
    for (let i = deck.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [deck[i], deck[j]] = [deck[j], deck[i]];
    }
    
    dispatch(gameActions.setDeck({ cardType, deck }));
  }
}
```

```typescript
// services/BankingService.ts

class BankingService {
  /**
   * Transfer money between players
   */
  transfer(
    fromPlayerId: string,
    toPlayerId: string,
    amount: number,
    dispatch: AppDispatch
  ): GameResult {
    const state = store.getState();
    const fromPlayer = state.players.entities[fromPlayerId];
    
    if (!fromPlayer) {
      return { success: false, error: 'Player not found' };
    }
    
    if (fromPlayer.cash < amount) {
      return { success: false, error: 'Insufficient funds' };
    }
    
    dispatch(updateCash({ playerId: fromPlayerId, amount: -amount }));
    dispatch(updateCash({ playerId: toPlayerId, amount }));
    
    return { success: true };
  }
  
  /**
   * Collect salary (passing GO)
   */
  collectSalary(playerId: string, dispatch: AppDispatch): void {
    const state = store.getState();
    const salaryAmount = state.game.config.salaryAmount;
    
    dispatch(updateCash({ playerId, amount: salaryAmount }));
    dispatch(updateStats({
      playerId,
      stats: { totalMoneyEarned: salaryAmount },
    }));
  }
  
  /**
   * Pay rent
   */
  payRent(
    fromPlayerId: string,
    toPlayerId: string,
    propertyId: string,
    diceRoll?: number,
    dispatch: AppDispatch
  ): GameResult {
    const state = store.getState();
    const property = state.properties.entities[propertyId];
    
    if (!property || property.isMortgaged) {
      return { success: false, error: 'Property is mortgaged' };
    }
    
    const rentAmount = this.calculateRent(property, toPlayerId, diceRoll, state);
    
    const result = this.transfer(fromPlayerId, toPlayerId, rentAmount, dispatch);
    
    if (result.success) {
      dispatch(updateStats({
        playerId: fromPlayerId,
        stats: { totalRentPaid: rentAmount },
      }));
      
      dispatch(updateStats({
        playerId: toPlayerId,
        stats: { totalRentCollected: rentAmount },
      }));
    }
    
    return result;
  }
  
  /**
   * Calculate rent for a property
   */
  private calculateRent(
    property: Property,
    ownerId: string,
    diceRoll: number | undefined,
    state: RootState
  ): number {
    if (property.type === PropertyType.STREET) {
      const streetProp = property as StreetProperty;
      const monopolies = selectPlayerMonopolies(ownerId)(state);
      const hasMonopoly = monopolies.includes(streetProp.colorGroup);
      
      if (streetProp.buildings === 0) {
        return hasMonopoly ? streetProp.rentWithMonopoly : streetProp.baseRent;
      } else if (streetProp.buildings === 1) {
        return streetProp.rentWith1House;
      } else if (streetProp.buildings === 2) {
        return streetProp.rentWith2Houses;
      } else if (streetProp.buildings === 3) {
        return streetProp.rentWith3Houses;
      } else if (streetProp.buildings === 4) {
        return streetProp.rentWith4Houses;
      } else {
        return streetProp.rentWithHotel;
      }
    } else if (property.type === PropertyType.RAILROAD) {
      const railroadProp = property as RailroadProperty;
      const ownedRailroads = propertiesSelectors
        .selectAll(state)
        .filter(p => p.type === PropertyType.RAILROAD && p.ownerId === ownerId)
        .length;
      
      switch (ownedRailroads) {
        case 1: return railroadProp.rent1Railroad;
        case 2: return railroadProp.rent2Railroads;
        case 3: return railroadProp.rent3Railroads;
        case 4: return railroadProp.rent4Railroads;
        default: return 0;
      }
    } else if (property.type === PropertyType.UTILITY) {
      const utilityProp = property as UtilityProperty;
      const ownedUtilities = propertiesSelectors
        .selectAll(state)
        .filter(p => p.type === PropertyType.UTILITY && p.ownerId === ownerId)
        .length;
      
      const multiplier = ownedUtilities === 1 
        ? utilityProp.multiplier1Utility 
        : utilityProp.multiplier2Utilities;
      
      return (diceRoll || 0) * multiplier;
    }
    
    return 0;
  }
  
  /**
   * Calculate player's total assets
   */
  calculateNetWorth(playerId: string, state: RootState): number {
    const player = state.players.entities[playerId];
    if (!player) return 0;
    
    let netWorth = player.cash;
    
    // Add property values
    player.propertyIds.forEach(propId => {
      const property = state.properties.entities[propId];
      if (property) {
        if (property.isMortgaged) {
          // Mortgaged properties contribute mortgage value minus unmortgage cost
          netWorth += property.mortgageValue * 0.9;
        } else {
          netWorth += property.price;
        }
        
        // Add building values
        if (property.type === PropertyType.STREET) {
          const streetProp = property as StreetProperty;
          if (streetProp.buildings > 0) {
            const buildingValue = streetProp.buildings * streetProp.houseCost * 0.5;
            netWorth += buildingValue;
          }
        }
      }
    });
    
    // Add Get Out of Jail cards
    netWorth += player.getOutOfJailCards * 50;
    
    return netWorth;
  }
}
```

```typescript
// services/BankruptcyService.ts

class BankruptcyService {
  /**
   * Check if player can pay debt
   */
  canPayDebt(playerId: string, amount: number, state: RootState): boolean {
    const player = state.players.entities[playerId];
    if (!player) return false;
    
    // Calculate maximum possible liquid assets
    let availableCash = player.cash;
    
    // Add mortgageable properties
    player.propertyIds.forEach(propId => {
      const property = state.properties.entities[propId];
      if (property && !property.isMortgaged) {
        availableCash += property.mortgageValue;
      }
    });
    
    // Add sellable buildings
    player.propertyIds.forEach(propId => {
      const property = state.properties.entities[propId] as StreetProperty;
      if (property && property.type === PropertyType.STREET && property.buildings > 0) {
        availableCash += property.buildings * property.houseCost * 0.5;
      }
    });
    
    return availableCash >= amount;
  }
  
  /**
   * Attempt to raise funds
   */
  raiseFunds(
    playerId: string,
    targetAmount: number,
    dispatch: AppDispatch
  ): GameResult {
    const state = store.getState();
    const player = state.players.entities[playerId];
    
    if (!player) {
      return { success: false, error: 'Player not found' };
    }
    
    let raisedAmount = 0;
    const actions: string[] = [];
    
    // Strategy: Sell buildings first, then mortgage properties
    
    // 1. Sell buildings
    const propertiesWithBuildings = player.propertyIds
      .map(id => state.properties.entities[id] as StreetProperty)
      .filter(p => p && p.type === PropertyType.STREET && p.buildings > 0)
      .sort((a, b) => {
        // Sell hotels before houses, expensive before cheap
        if (a.buildings === 5 && b.buildings !== 5) return -1;
        if (a.buildings !== 5 && b.buildings === 5) return 1;
        return b.houseCost - a.houseCost;
      });
    
    for (const property of propertiesWithBuildings) {
      if (player.cash + raisedAmount >= targetAmount) break;
      
      const buildingsToSell = property.buildings;
      const value = buildingsToSell * property.houseCost * 0.5;
      
      dispatch(removeBuilding({ propertyId: property.id, count: buildingsToSell }));
      raisedAmount += value;
      actions.push(`Sold ${buildingsToSell} buildings from ${property.name}`);
    }
    
    // 2. Mortgage properties
    if (player.cash + raisedAmount < targetAmount) {
      const unmortgagedProperties = player.propertyIds
        .map(id => state.properties.entities[id])
        .filter(p => p && !p.isMortgaged)
        .sort((a, b) => a!.mortgageValue - b!.mortgageValue); // Mortgage cheap first
      
      for (const property of unmortgagedProperties) {
        if (player.cash + raisedAmount >= targetAmount) break;
        
        if (property) {
          dispatch(mortgageProperty(property.id));
          raisedAmount += property.mortgageValue;
          actions.push(`Mortgaged ${property.name}`);
        }
      }
    }
    
    // Update player cash
    dispatch(updateCash({ playerId, amount: raisedAmount }));
    
    if (player.cash + raisedAmount >= targetAmount) {
      return {
        success: true,
        data: { raisedAmount, actions },
      };
    } else {
      return {
        success: false,
        error: 'Cannot raise sufficient funds',
        data: { raisedAmount, actions, shortfall: targetAmount - (player.cash + raisedAmount) },
      };
    }
  }
  
  /**
   * Declare bankruptcy
   */
  declareBankruptcy(
    playerId: string,
    creditorId: string | 'BANK',
    dispatch: AppDispatch
  ): void {
    const state = store.getState();
    const player = state.players.entities[playerId];
    
    if (!player) return;
    
    // Mark player as bankrupt
    dispatch(declareBankruptcy({ playerId, turn: state.game.turn }));
    
    // Transfer assets
    if (creditorId === 'BANK') {
      // Return all properties to bank
      player.propertyIds.forEach(propId => {
        const property = state.properties.entities[propId];
        if (property) {
          // Remove buildings
          if (property.type === PropertyType.STREET) {
            const streetProp = property as StreetProperty;
            if (streetProp.buildings > 0) {
              dispatch(removeBuilding({ propertyId: propId, count: streetProp.buildings }));
            }
          }
          
          // Unmortgage if needed
          if (property.isMortgaged) {
            dispatch(unmortgageProperty(propId));
          }
          
          // Clear owner
          dispatch(clearOwner(propId));
        }
      });
    } else {
      // Transfer everything to creditor
      player.propertyIds.forEach(propId => {
        dispatch(transferProperty({
          propertyId: propId,
          fromPlayerId: playerId,
          toPlayerId: creditorId,
        }));
        
        dispatch(addProperty({ playerId: creditorId, propertyId: propId }));
      });
      
      // Transfer cash
      if (player.cash > 0) {
        dispatch(updateCash({ playerId: creditorId, amount: player.cash }));
      }
      
      // Transfer Get Out of Jail cards
      if (player.getOutOfJailCards > 0) {
        dispatch(playersActions.transferGetOutOfJailCards({
          fromPlayerId: playerId,
          toPlayerId: creditorId,
          count: player.getOutOfJailCards,
        }));
      }
    }
    
    // Clear player properties
    dispatch(playersActions.clearProperties(playerId));
    
    // Check if game should end
    const activePlayers = state.players.ids.filter(
      id => !state.players.entities[id].isBankrupt
    );
    
    if (activePlayers.length === 1) {
      dispatch(gameActions.endGame({
        winnerId: activePlayers[0],
        endTime: Date.now(),
      }));
    }
  }
}
```

```typescript
// lib/socket.ts

import { io, Socket } from 'socket.io-client';

class SocketService {
  private socket: Socket | null = null;
  private static instance: SocketService;
  
  private constructor() {}
  
  static getInstance(): SocketService {
    if (!SocketService.instance) {
      SocketService.instance = new SocketService();
    }
    return SocketService.instance;
  }
  
  connect(serverUrl: string): void {
    this.socket = io(serverUrl, {
      transports: ['websocket'],
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionAttempts: 5,
    });
    
    this.setupListeners();
  }
  
  private setupListeners(): void {
    if (!this.socket) return;
    
    this.socket.on('connect', () => {
      console.log('Connected to server');
    });
    
    this.socket.on('disconnect', () => {
      console.log('Disconnected from server');
    });
    
    this.socket.on('error', (error) => {
      console.error('Socket error:', error);
    });
  }
  
  emit(event: string, data: any): void {
    if (this.socket) {
      this.socket.emit(event, data);
    }
  }
  
  on(event: string, callback: (data: any) => void): void {
    if (this.socket) {
      this.socket.on(event, callback);
    }
  }
  
  off(event: string): void {
    if (this.socket) {
      this.socket.off(event);
    }
  }
  
  disconnect(): void {
    if (this.socket) {
      this.socket.disconnect();
      this.socket = null;
    }
  }
}

export default SocketService;
```

```typescript
// utils/soundManager.ts

class SoundManager {
  private sounds: Map<string, HTMLAudioElement> = new Map();
  private music: HTMLAudioElement | null = null;
  private soundEnabled = true;
  private musicEnabled = true;
  private volume = 0.7;
  
  loadSound(name: string, url: string): void {
    const audio = new Audio(url);
    audio.volume = this.volume;
    this.sounds.set(name, audio);
  }
  
  playSound(name: string): void {
    if (!this.soundEnabled) return;
    
    const sound = this.sounds.get(name);
    if (sound) {
      sound.currentTime = 0;
      sound.play().catch(console.error);
    }
  }
  
  playMusic(url: string, loop = true): void {
    if (!this.musicEnabled) return;
    
    if (this.music) {
      this.music.pause();
    }
    
    this.music = new Audio(url);
    this.music.loop = loop;
    this.music.volume = this.volume * 0.3; // Lower volume for background music
    this.music.play().catch(console.error);
  }
  
  stopMusic(): void {
    if (this.music) {
      this.music.pause();
      this.music = null;
    }
  }
  
  setVolume(volume: number): void {
    this.volume = Math.max(0, Math.min(1, volume));
    this.sounds.forEach(sound => {
      sound.volume = this.volume;
    });
    if (this.music) {
      this.music.volume = this.volume * 0.3;
    }
  }
  
  toggleSound(enabled?: boolean): void {
    this.soundEnabled = enabled ?? !this.soundEnabled;
  }
  
  toggleMusic(enabled?: boolean): void {
    this.musicEnabled = enabled ?? !this.musicEnabled;
    if (!this.musicEnabled && this.music) {
      this.stopMusic();
    }
  }
}

export default new SoundManager();
```

```typescript
// services/SaveLoadService.ts

class SaveLoadService {
  private readonly STORAGE_KEY = 'monopoly_save';
  
  /**
   * Save game state
   */
  saveGame(state: RootState): SaveResult {
    try {
      const saveData: SaveData = {
        version: '1.0.0',
        timestamp: Date.now(),
        game: state.game,
        players: state.players,
        properties: state.properties,
        notifications: {
          history: state.notifications.history,
        },
      };
      
      const serialized = JSON.stringify(saveData);
      localStorage.setItem(this.STORAGE_KEY, serialized);
      
      return { success: true, savedAt: saveData.timestamp };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : 'Unknown error',
      };
    }
  }
  
  /**
   * Load game state
   */
  loadGame(): LoadResult {
    try {
      const serialized = localStorage.getItem(this.STORAGE_KEY);
      
      if (!serialized) {
        return { success: false, error: 'No saved game found' };
      }
      
      const saveData: SaveData = JSON.parse(serialized);
      
      // Validate save data
      if (!this.validateSaveData(saveData)) {
        return { success: false, error: 'Invalid save data' };
      }
      
      return {
        success: true,
        data: saveData,
      };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : 'Unknown error',
      };
    }
  }
  
  /**
   * Export game to file
   */
  exportGame(state: RootState): void {
    const saveData: SaveData = {
      version: '1.0.0',
      timestamp: Date.now(),
      game: state.game,
      players: state.players,
      properties: state.properties,
      notifications: {
        history: state.notifications.history,
      },
    };
    
    const blob = new Blob([JSON.stringify(saveData, null, 2)], {
      type: 'application/json',
    });
    
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `monopoly-save-${Date.now()}.json`;
    a.click();
    
    URL.revokeObjectURL(url);
  }
  
  /**
   * Import game from file
   */
  async importGame(file: File): Promise<LoadResult> {
    try {
      const text = await file.text();
      const saveData: SaveData = JSON.parse(text);
      
      if (!this.validateSaveData(saveData)) {
        return { success: false, error: 'Invalid save file' };
      }
      
      return {
        success: true,
        data: saveData,
      };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : 'Unknown error',
      };
    }
  }
  
  private validateSaveData(data: any): data is SaveData {
    return (
      data &&
      typeof data === 'object' &&
      'version' in data &&
      'timestamp' in data &&
      'game' in data &&
      'players' in data &&
      'properties' in data
    );
  }
}
```

```typescript
// services/StatsService.ts

class StatsService {
  /**
   * Calculate game statistics
   */
  calculateGameStats(state: RootState): GameStatistics {
    const players = playersSelectors.selectAll(state);
    const activePlayers = players.filter(p => !p.isBankrupt);
    
    return {
      gameDuration: Date.now() - (state.game.createdAt || 0),
      totalTurns: state.game.turn,
      totalRounds: state.game.round,
      
      playerStats: players.map(player => ({
        playerId: player.id,
        playerName: player.name,
        finalPosition: this.calculateFinalPosition(player, state),
        netWorth: this.calculateNetWorth(player.id, state),
        ...player.stats,
      })),
      
      propertyStats: {
        totalPropertiesOwned: propertiesSelectors
          .selectAll(state)
          .filter(p => p.ownerId).length,
        totalMonopolies: this.countTotalMonopolies(state),
        totalBuildings: this.countTotalBuildings(state),
      },
      
      economicStats: {
        totalMoneyInCirculation: players.reduce((sum, p) => sum + p.cash, 0),
        totalPropertyValue: this.calculateTotalPropertyValue(state),
        averageNetWorth: players.reduce((sum, p) => 
          sum + this.calculateNetWorth(p.id, state), 0) / players.length,
      },
    };
  }
  
  /**
   * Generate game report
   */
  generateGameReport(state: RootState): GameReport {
    const stats = this.calculateGameStats(state);
    const winner = state.game.winnerId 
      ? state.players.entities[state.game.winnerId]
      : null;
    
    return {
      summary: {
        winner: winner?.name || 'N/A',
        duration: this.formatDuration(stats.gameDuration),
        totalTurns: stats.totalTurns,
        totalRounds: stats.totalRounds,
      },
      
      playerRankings: stats.playerStats
        .sort((a, b) => b.netWorth - a.netWorth)
        .map((p, index) => ({
          rank: index + 1,
          playerName: p.playerName,
          netWorth: p.netWorth,
          propertiesOwned: p.propertiesPurchased,
          totalRentCollected: p.totalRentCollected,
        })),
      
      highlights: this.generateHighlights(stats),
    };
  }
  
  private generateHighlights(stats: GameStatistics): string[] {
    const highlights: string[] = [];
    
    // Richest player
    const richest = stats.playerStats.reduce((max, p) => 
      p.netWorth > max.netWorth ? p : max
    );
    highlights.push(`${richest.playerName} was the richest with $${richest.netWorth}`);
    
    // Most properties
    const mostProperties = stats.playerStats.reduce((max, p) =>
      p.propertiesPurchased > max.propertiesPurchased ? p : max
    );
    highlights.push(`${mostProperties.playerName} owned the most properties (${mostProperties.propertiesPurchased})`);
    
    // Most rent collected
    const mostRent = stats.playerStats.reduce((max, p) =>
      p.totalRentCollected > max.totalRentCollected ? p : max
    );
    highlights.push(`${mostRent.playerName} collected the most rent ($${mostRent.totalRentCollected})`);
    
    return highlights;
  }
}
```
